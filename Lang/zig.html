Pengenalan Zig Language

Zig adalah bahasa pemrograman tujuan umum yang dirancang untuk robustness, optimalitas, dan maintainability. Zig menawarkan alternatif yang lebih sederhana dan aman dibandingkan C sambil mempertahankan performa yang tinggi.

Performa Tinggi

Kompilasi langsung ke machine code tanpa overhead

Memory Safety

Fitur built-in untuk mencegah memory leaks

Sederhana

Sintaks yang bersih dan mudah dipahami

Instalasi Zig

Metode Instalasi

1. Menggunakan Package Manager

# macOS (Homebrew)
brew install zig

# Ubuntu/Debian
sudo apt install zig

# Arch Linux
sudo pacman -S zig

# Windows (Scoop)
scoop install zig

2. Download Manual

Download dari website resmi:

ziglang.org/download

3. Verifikasi Instalasi

zig version

Hello World

Mari kita mulai dengan program Hello World klasik dalam Zig:

const std = @import("std");

pub fn main() !void {
    std.debug.print("Hello, World!\n", .{});
}

Penjelasan Kode:

  • const std = @import("std") - Mengimpor standard library
  • pub fn main() !void - Fungsi utama yang mengembalikan error union
  • std.debug.print - Fungsi untuk mencetak output ke console
  • .{} - Argumen kosong karena tidak ada format specifier

Menjalankan Program:

# Simpan sebagai hello.zig
zig run hello.zig

Variabel

Zig memiliki dua cara untuk mendeklarasikan variabel: const dan var.

Constants (const)

const std = @import("std");

pub fn main() void {
    const PI = 3.14159;
    const language = "Zig";
    
    std.debug.print("PI = {d}\n", .{PI});
    std.debug.print("Language = {s}\n", .{language});
}

const membuat variabel immutable (tidak bisa diubah setelah deklarasi)

Variables (var)

const std = @import("std");

pub fn main() void {
    var counter: i32 = 0;
    counter = counter + 1;
    
    var message: []const u8 = "Hello";
    message = "Hello, Zig!";
    
    std.debug.print("Counter = {}\n", .{counter});
    std.debug.print("Message = {s}\n", .{message});
}

var membuat variabel mutable (bisa diubah)

Type Inference:

Zig bisa menebak tipe data secara otomatis, tapi Anda juga bisa menentukannya secara eksplisit:

// Type inference
const name = "Alice"; // []const u8
const age = 25;       // i32

// Explicit type
const height: f32 = 1.75;
const is_active: bool = true;

Tipe Data

Tipe Data Primitif

// Integers
const a: i8 = -128;      // 8-bit signed
const b: u8 = 255;       // 8-bit unsigned
const c: i32 = 1000;     // 32-bit signed
const d: u64 = 1000000;  // 64-bit unsigned

// Floats
const pi: f32 = 3.14159; // 32-bit float
const e: f64 = 2.71828;  // 64-bit float

// Boolean
const is_true: bool = true;
const is_false: bool = false;

// Characters
const letter: u8 = 'A';

Tipe Data Komposit

// Arrays
const numbers = [5]i32{ 1, 2, 3, 4, 5 };

// Slices
const slice = numbers[0..3];

// Strings (slices of u8)
const greeting = "Hello, Zig!";

// Pointers
var value: i32 = 42;
const ptr = &value;

// Optionals
const maybe_number: ?i32 = null;
const definite_number: ?i32 = 123;

Contoh Penggunaan

const std = @import("std");

pub fn main() void {
    // Array operations
    const scores = [4]i32{ 85, 92, 78, 96 };
    const average = @divExact(scores[0] + scores[1] + scores[2] + scores[3], 4);
    
    // String operations
    const name = "Zig Programmer";
    const name_length = name.len;
    
    // Optional handling
    const optional_value: ?i32 = 42;
    if (optional_value) |value| {
        std.debug.print("Value exists: {}\n", .{value});
    } else {
        std.debug.print("Value is null\n", .{});
    }
    
    std.debug.print("Average score: {}\n", .{average});
    std.debug.print("Name length: {}\n", .{name_length});
}

Fungsi

Deklarasi Fungsi Dasar

const std = @import("std");

// Fungsi tanpa parameter dan return value
fn greet() void {
    std.debug.print("Hello from Zig!\n", .{});
}

// Fungsi dengan parameter
fn add(a: i32, b: i32) i32 {
    return a + b;
}

// Fungsi dengan error handling
fn divide(a: f32, b: f32) !f32 {
    if (b == 0) {
        return error.DivisionByZero;
    }
    return a / b;
}

pub fn main() !void {
    greet();
    
    const sum = add(5, 3);
    std.debug.print("5 + 3 = {}\n", .{sum});
    
    const result = divide(10.0, 2.0) catch |err| {
        std.debug.print("Error: {}\n", .{err});
        return;
    };
    std.debug.print("10 / 2 = {d}\n", .{result});
}

Fungsi dengan Multiple Return Values

const std = @import("std");

// Mengembalikan struct
fn minMax(numbers: []const i32) struct { min: i32, max: i32 } {
    var min_val = numbers[0];
    var max_val = numbers[0];
    
    for (numbers) |num| {
        if (num < min_val) min_val = num;
        if (num > max_val) max_val = num;
    }
    
    return .{ .min = min_val, .max = max_val };
}

pub fn main() void {
    const data = [5]i32{ 10, 5, 20, 15, 8 };
    const result = minMax(&data);
    
    std.debug.print("Min: {}, Max: {}\n", .{ result.min, result.max });
}

Tips Fungsi:

  • Gunakan pub untuk membuat fungsi bisa diakses dari file lain
  • ! sebelum return type menandakan fungsi bisa mengembalikan error
  • Gunakan catch untuk menangani error
  • Fungsi bisa mengembalikan multiple values menggunakan struct atau tuple

Kontrol Alur

If-Else

const std = @import("std");

pub fn main() void {
    const age = 18;
    
    if (age >= 18) {
        std.debug.print("Dewasa\n", .{});
    } else {
        std.debug.print("Belum dewasa\n", .{});
    }
    
    // If sebagai expression
    const message = if (age >= 18) "Bisa memilih" else "Belum bisa memilih";
    std.debug.print("{s}\n", .{message});
}

Switch

const std = @import("std");

pub fn main() void {
    const grade = 'B';
    
    switch (grade) {
        'A' => std.debug.print("Excellent!\n", .{}),
        'B' => std.debug.print("Good job!\n", .{}),
        'C' => std.debug.print("Pass\n", .{}),
        'D', 'F' => std.debug.print("Need improvement\n", .{}),
        else => std.debug.print("Invalid grade\n", .{}),
    }
    
    // Switch sebagai expression
    const score = switch (grade) {
        'A' => 90,
        'B' => 80,
        'C' => 70,
        else => 60,
    };
    std.debug.print("Score: {}\n", .{score});
}

Loops

const std = @import("std");

pub fn main() void {
    // While loop
    var i: i32 = 0;
    while (i < 5) : (i += 1) {
        std.debug.print("While: {}\n", .{i});
    }
    
    // For loop
    const numbers = [5]i32{ 10, 20, 30, 40, 50 };
    for (numbers) |num| {
        std.debug.print("For: {}\n", .{num});
    }
    
    // For dengan index
    for (numbers, 0..) |num, index| {
        std.debug.print("Index {}: {}\n", .{ index, num });
    }
    
    // Break dan continue
    var j: i32 = 0;
    while (j < 10) : (j += 1) {
        if (j == 3) continue;
        if (j == 7) break;
        std.debug.print("Loop: {}\n", .{j});
    }
}

Defer dan Errdefer

const std = @import("std");

pub fn main() void {
    // Defer - dieksekusi saat keluar dari scope
    {
        std.debug.print("Before defer\n", .{});
        defer {
            std.debug.print("Deferred execution\n", .{});
        }
        std.debug.print("After defer\n", .{});
    }
    
    // Errdefer - dieksekusi hanya saat terjadi error
    const result = riskyOperation() catch |err| {
        std.debug.print("Caught error: {}\n", .{err});
        return;
    };
    std.debug.print("Operation succeeded: {}\n", .{result});
}

fn riskyOperation() !i32 {
    errdefer {
        std.debug.print("Cleaning up after error\n", .{});
    }
    
    // Simulate error
    // return error.SomethingWentWrong;
    
    return 42;
}

Structs

Deklarasi Struct Dasar

const std = @import("std");

const Person = struct {
    name: []const u8,
    age: i32,
    height: f32,
    
    // Method
    fn greet(self: *const Person) void {
        std.debug.print("Hello, my name is {s}!\n", .{self.name});
    }
    
    // Static method
    fn create(name: []const u8, age: i32, height: f32) Person {
        return Person{
            .name = name,
            .age = age,
            .height = height,
        };
    }
};

pub fn main() void {
    // Membuat instance
    var person = Person{
        .name = "Alice",
        .age = 25,
        .height = 1.65,
    };
    
    // Mengakses fields
    std.debug.print("Name: {s}, Age: {}\n", .{ person.name, person.age });
    
    // Memanggil method
    person.greet();
    
    // Menggunakan static method
    const person2 = Person.create("Bob", 30, 1.75);
    person2.greet();
}

Struct dengan Methods

const std = @import("std");

const Rectangle = struct {
    width: f32,
    height: f32,
    
    const Self = @This();
    
    fn area(self: Self) f32 {
        return self.width * self.height;
    }
    
    fn perimeter(self: Self) f32 {
        return 2 * (self.width + self.height);
    }
    
    fn isSquare(self: Self) bool {
        return self.width == self.height;
    }
    
    fn scale(self: *Self, factor: f32) void {
        self.width *= factor;
        self.height *= factor;
    }
};

pub fn main() void {
    var rect = Rectangle{
        .width = 10.0,
        .height = 5.0,
    };
    
    std.debug.print("Area: {d}\n", .{rect.area()});
    std.debug.print("Perimeter: {d}\n", .{rect.perimeter()});
    std.debug.print("Is square: {}\n", .{rect.isSquare()});
    
    rect.scale(2.0);
    std.debug.print("After scaling - Width: {d}, Height: {d}\n", .{ rect.width, rect.height });
}

Manajemen Memori

Allocator

const std = @import("std");

pub fn main() !void {
    // General Purpose Allocator
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    const allocator = gpa.allocator();
    
    // Alokasi memori
    const memory = try allocator.alloc(u8, 100);
    defer allocator.free(memory);
    
    std.debug.print("Allocated {} bytes\n", .{memory.len});
    
    // Menggunakan memori
    for (memory, 0..) |*byte, i| {
        byte.* = @intCast(i % 256);
    }
    
    std.debug.print("First 10 bytes: ", .{});
    for (memory[0..10]) |byte| {
        std.debug.print("{d} ", .{byte});
    }
    std.debug.print("\n", .{});
}

Arena Allocator

const std = @import("std");

pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    
    // Arena allocator untuk deallokasi massal
    var arena = std.heap.ArenaAllocator.init(gpa.allocator());
    defer arena.deinit();
    const allocator = arena.allocator();
    
    // Alokasi beberapa blok memori
    const block1 = try allocator.alloc(u8, 50);
    const block2 = try allocator.alloc(i32, 25);
    const block3 = try allocator.alloc(f64, 10);
    
    // Tidak perlu free individual, arena.deinit() akan membersihkan semua
    std.debug.print("Allocated blocks: {} bytes, {} ints, {} floats\n", .{
        block1.len, block2.len, block3.len
    });
}

String Allocation

const std = @import("std");

pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    const allocator = gpa.allocator();
    
    // Membuat string di heap
    const name = try allocator.dupe(u8, "Zig Programmer");
    defer allocator.free(name);
    
    std.debug.print("Name: {s}\n", .{name});
    
    // Concatenate strings
    const greeting = try std.fmt.allocPrint(allocator, "Hello, {s}!", .{name});
    defer allocator.free(greeting);
    
    std.debug.print("{s}\n", .{greeting});
    
    // Dynamic string building
    var buffer = std.ArrayList(u8).init(allocator);
    defer buffer.deinit();
    
    try buffer.append('H');
    try buffer.appendSlice("ello ");
    try buffer.appendSlice("World!");
    
    std.debug.print("{s}\n", .{buffer.items});
}

Penting:

  • Selalu gunakan defer untuk memastikan memori dibebaskan
  • Gunakan try untuk menangani error allocation
  • Arena allocator berguna untuk alokasi massal dengan lifetime sama
  • General Purpose Allocator (GPA) adalah allocator default yang fleksibel

Konkurensi

Async/Await

const std = @import("std");

pub fn main() !void {
    // Membuat frame untuk async operations
    var frame = async fibonacci(10);
    const result = await frame;
    
    std.debug.print("Fibonacci(10) = {}\n", .{result});
}

fn fibonacci(n: u32) u32 {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

Multiple Async Operations

const std = @import("std");

pub fn main() !void {
    // Menjalankan multiple async operations
    const frame1 = async computeTask(1, 100);
    const frame2 = async computeTask(2, 200);
    const frame3 = async computeTask(3, 150);
    
    // Menunggu semua selesai
    const result1 = await frame1;
    const result2 = await frame2;
    const result3 = await frame3;
    
    std.debug.print("Results: {}, {}, {}\n", .{ result1, result2, result3 });
}

fn computeTask(id: u32, delay_ms: u32) u32 {
    // Simulasi work
    std.time.sleep(delay_ms * std.time.ns_per_ms);
    return id * 100;
}

Thread Pool

const std = @import("std");

pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    const allocator = gpa.allocator();
    
    // Membuat thread pool
    var pool: std.Thread.Pool = undefined;
    try pool.init(.{ .allocator = allocator });
    defer pool.deinit();
    
    // Shared data
    const counter = std.atomic.Atomic(u32).init(0);
    
    // Menjalankan tasks di thread pool
    var i: u32 = 0;
    while (i < 10) : (i += 1) {
        try pool.spawn(worker, .{ &counter, i });
    }
    
    // Menunggu semua tasks selesai
    pool.wait();
    
    std.debug.print("Final counter: {}\n", .{counter.load(.Monotonic)});
}

fn worker(counter: *std.atomic.Atomic(u32), id: u32) void {
    // Simulasi work
    std.time.sleep(100 * std.time.ns_per_ms);
    
    // Atomic increment
    _ = counter.fetchAdd(1, .Monotonic);
    
    std.debug.print("Worker {} completed\n", .{id});
}

Fitur Konkurensi Zig:

  • async/await untuk asynchronous programming
  • Thread pool untuk parallel execution
  • Atomic operations untuk thread-safe data access
  • No data races by design (compile-time checks)
  • Minimal runtime overhead

File I/O

const std = @import("std");

pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    const allocator = gpa.allocator();
    
    // Menulis ke file
    const file = try std.fs.cwd().createFile("hello.txt", .{});
    defer file.close();
    
    try file.writeAll("Hello, Zig!\nThis is a test file.");
    
    // Membaca dari file
    const contents = try file.reader().readAllAlloc(allocator, 1024);
    defer allocator.free(contents);
    
    std.debug.print("File contents:\n{s}\n", .{contents});
}

JSON Parsing

const std = @import("std");

pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    const allocator = gpa.allocator();
    
    const json_str = 
        \\{
        \\  "name": "Zig",
        \\  "version": "0.11.0",
        \\  "features": ["fast", "simple", "safe"]
        \\}
    ;
    
    var parsed = try std.json.parseFromSlice(
        std.json.Value,
        allocator,
        json_str,
        .{},
    );
    defer parsed.deinit();
    
    const root = parsed.value;
    
    if (root.object.get("name")) |name| {
        std.debug.print("Language: {s}\n", .{name.string});
    }
    
    if (root.object.get("features")) |features| {
        std.debug.print("Features:\n", .{});
        for (features.array.items) |feature| {
            std.debug.print("  - {s}\n", .{feature.string});
        }
    }
}

HTTP Server

const std = @import("std");

pub fn main() !void {
    var gpa = std.heap.GeneralPurposeAllocator(.{}){};
    defer _ = gpa.deinit();
    const allocator = gpa.allocator();
    
    // Simple HTTP server
    const address = try std.net.Address.parseIp("127.0.0.1", 8080);
    var server = address.listen(.{ .reuse_port = true }) catch |err| {
        std.debug.print("Failed to listen: {}\n", .{err});
        return;
    };
    defer server.deinit();
    
    std.debug.print("Server running on http://127.0.0.1:8080\n", .{});
    
    while (true) {
        const connection = server.accept() catch |err| {
            std.debug.print("Accept error: {}\n", .{err});
            continue;
        };
        defer connection.stream.close();
        
        handleConnection(allocator, connection.stream) catch |err| {
            std.debug.print("Connection error: {}\n", .{err});
        };
    }
}

fn handleConnection(allocator: std.mem.Allocator, stream: std.net.Stream) !void {
    var buffer: [1024]u8 = undefined;
    const bytes_read = try stream.read(&buffer);
    const request = buffer[0..bytes_read];
    
    // Simple response
    const response = 
        \\HTTP/1.1 200 OK
        \\Content-Type: text/html
        \\Connection: close
        \\
        \\
        \\
        \\
Hello from Zig!

        \\
This is a simple HTTP server.

        \\
        \\
    ;
    
    _ = try stream.writeAll(response);
}
